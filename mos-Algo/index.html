#include<stdio.h>
#include<math.h>
#include<algorithm>
#define reg register
#define ri reg int
#define rep(i, x, y) for(ri i = x; i <= y; ++i)
#define nrep(i, x, y) for(ri i = x; i >= y; --i)
#define DEBUG 1
#define ll long long
#define max(i, j) (i) > (j) ? (i) : (j)
#define min(i, j) (i) < (j) ? (i) : (j)
using namespace std;
int read(){
    int x = 0, w = 1;
	char ch = getchar();
    while(ch < '0' || ch > '9') { if(ch == '-') w = -1; ch = getchar(); }
    while(ch >= '0' && ch <= '9') x = x * 10 + ch - '0', ch = getchar();
    return x * w;
}
struct IO {
	#define MAXSIZE (1 << 20)
	#define isdigit(x) (x >= '0' && x <= '9')
	char buf[MAXSIZE], *p1, *p2;
	char pbuf[MAXSIZE], *pp;
	#if DEBUG
	#else
		IO() : p1(buf), p2(buf), pp(pbuf) {}
		~IO() {
			fwrite(pbuf, 1, pp - pbuf, stdout);
		}
	#endif
	inline char gc() {
		#if DEBUG
    		return getchar();
		#endif
    	if(p1 == p2)
			p2 = (p1 = buf) + fread(buf, 1, MAXSIZE, stdin);
    	return p1 == p2 ? ' ' : *p1++;
	}
	inline bool blank(char ch) {
    	return ch == ' ' || ch == '\n' || ch == '\r' || ch == '\t';
	}
	template <class T>
	inline void read(T &x) {
    	register double tmp = 1;
    	register bool sign = 0;
    	x = 0;
    	register char ch = gc();
    	for(; !isdigit(ch); ch = gc())
    		if(ch == '-') sign = 1;
    	for(; isdigit(ch); ch = gc())
			x = x * 10 + (ch - '0');
    	if(ch == '.')
			for(ch = gc(); isdigit(ch); ch = gc())
        		tmp /= 10.0, x += tmp * (ch - '0');
    	if(sign) x = -x;
	}
	inline void read(char *s) {
		register char ch = gc();
		for(; blank(ch); ch = gc());
		for(; !blank(ch); ch = gc())
			*s++ = ch;
    	*s = 0;
	}
	inline void read(char &c) {
    	for(c = gc(); blank(c); c = gc());
	}
	inline void push(const char &c) {
		#if DEBUG
    		putchar(c);
		#else
    		if(pp - pbuf == MAXSIZE) {
				fwrite(pbuf, 1, MAXSIZE, stdout);
				pp = pbuf;
			}
    		*pp++ = c;
		#endif
	}
	template <class T>
	inline void write(T x) {
    	if(x < 0) {
			x = -x;
			push('-');
		}
    	static T sta[35];
    	T top = 0;
    	do{
			sta[top++] = x % 10;
			x /= 10;
    	}while(x);
    	while(top)
			push(sta[--top] + '0');
	}
	template <class T>
	inline void write(T x, char lastChar) {
    	write(x);
		push(lastChar);
	}
} io;
int n, m, B, len, Ans, a[200010], d[200010], b[200010], t[200010], cnt[200010], ans[200010];
struct Q { int l, r, id; } q[200010];
bool cmp(Q x, Q y) {
	if(b[x.l] == b[y.l])
		if(b[x.l] & 1) return x.r < y.r;
		else return x.r > y.r;
	return b[x.l] < b[y.l];
}
void add(int i) { /*区间增长*/
	t[cnt[a[i]]]--; /*去掉区间的众数,因为他即将加一*/
	t[++cnt[a[i]]]++;
	Ans = max(Ans, cnt[a[i]]);
}
void del(int i) { /*区间缩短*/
	t[cnt[a[i]]]--;
	if(cnt[a[i]] == Ans && !t[cnt[a[i]]]) --Ans; /*如果去掉以后总共的众数随着他减小*/
	t[--cnt[a[i]]]++;
}
int main() {
	n = read(), m = read();
	B = sqrt(n); /*确定复杂度*/
	rep(i, 1, n) {
		d[i] = a[i] = read();
		b[i] = (i - 1) / B + 1; /*离散化处理*/
	}
	sort(d + 1, d + n + 1);
	len = unique(d + 1, d + n + 1) - d - 1;
	rep(i, 1, n) a[i] = lower_bound(d + 1, d + len + 1, a[i]) - d;
	rep(i, 1, m) q[i] = (Q){read(), read(), i};
	sort(q + 1, q + m + 1, cmp);
	int l = 1, r = 0;
	rep(i, 1, m) {
		while(r < q[i].r) add(++r);
		while(r > q[i].r) del(r--);
		while(l < q[i].l) del(l++);
		while(l > q[i].l) add(--l);
		ans[q[i].id] = Ans;
	}
	rep(i, 1, m) io.write(-ans[i], '\n');
	return 0;
}

int aa[1000010], cnt[1000010], belong[1000010];
int n, m, size, bnum, now, ans[1000010];
struct query { /*询问*/
	int l, r, id;
} q[1000010];

inline int cmp(query a, query b) {
	return (belong[a.l] ^ belong[b.l]) ? belong[a.l] < belong[b.l] : ((belong[a.l] & 1) ? a.r < b.r : a.r > b.r); /*奇偶优化后的排序*/
}
int main() {
	io.read(n);
	rep(i, 1, n) io.read(aa[i]), belong[i] = (i - 1) / 1700 + 1;
	io.read(m);
	rep(i, 1, m) {
		io.read(q[i].l), io.read(q[i].r);
		q[i].id = i; /*读入每一个询问*/
	}
	std::sort(q + 1, q + m + 1, cmp);
	ri l = 1, r = 0;
	rep(i, 1, m) {
		while(l < q[i].l) now -= !--cnt[aa[l++]]; /*需要移动就移,在移的过程中更新答案*/
		while(l > q[i].l) now += !cnt[aa[--l]]++;
		while(r < q[i].r) now += !cnt[aa[++r]]++;
		while(r > q[i].r) now -= !--cnt[aa[r--]];
		ans[q[i].id] = now;
	}
	rep(i, 1, m) io.write(ans[i], '\n');
}
