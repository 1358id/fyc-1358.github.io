void dijkstra() {
	priority_queue<pair<int,int> > Q;
	for(int i = 0; i < n; i++) d[i] = INF;
	d[1] = 0;
	Q.push(make_pair(-d[0], 0));
	while(!Q.empty()) {
		int x = Q.top().second; Q.pop();
		if(vis[x]) continue;
		vis[x] = 1;
		
		for(int i=0;i<G[x].size();i++) {
			int y=G[x][i];
			if(d[y] > d[x] + w[x][y]){
				d[y] = d[x] + w[x][y];
				Q.push(make_pair(-d[y], y));
			}
		}
	}
}

void spfa() {
	for(int i = 0; i < n; i++) d[i] = INF;
	d[1] = 0;
	queue<int> Q;
	Q.push(1);
	vis[1] = 1;
	while(!Q.empty()) {
		int x = Q.front(); Q.pop();
		vis[x] = 0;
		
		for(int i=0;i<G[x].size();i++){
			int y = G[x][i];
			if(d[y] > d[x] + w[x][y]){
				d[y] = d[x] + w[x][y];
				if(!vis[y])
					Q.push(y), vis[y] = 1;
			}
		}
	}
}

struct Edge{
	int u, v, w;
};
Edge e[M];
bool cmp(const Edge& x, const Edge& y) {
	return x.w < y.w;
}
int fa[N];
int find(int x) {return x == fa[x] ? x : fa[x] = find(fa[x]);}
void kruskal() {
	int ans = 0;
	sort(e, e + m, cmp);
	for(int i=1;i<=n;i++) fa[i] = i;
	for(int i=0;i<m;i++) {
		int u = e[i].u, v = e[i].v;
		if(find(u) != find(v)) {
			ans += e[i].w;
			
			int _u = find(u), _v = find(v);
			fa[_u] = _v;
		}
	}
}

vector<int> G[N];
int deg[N];
queue<int> Q;
int topo[N], cnt;

void toposort() {
	for(int i=1;i<=n;i++) {
		for(int j=0;j<G[i].size();j++) {
			deg[G[i][j]] ++;
		}
	}
	for(int i=1;i<=n;i++)
		if(deg[i] == 0)
			Q.push(i);
	
	while(!Q.empty()) {
		int x = Q.front(); Q.pop();
		cnt ++;
		topo[cnt] = x;
		for(int i=0;i<G[x];size();i++) {
			int y=G[x][i];
			deg[y] --;
			if(deg[y] == 0) {
				Q.push(y);
			}
		}
	}
}
